Le vector colony doit toujours être trié par vitesse de parcours de la piste.
Utiliser un lambda pour définir la manière dont les pingouins sont comparés.

______________________________________________________

void Pingouin::ajouterAuGroupe(shared_ptr<Pingouin> nouveauPingouin) {
    // Ajouter le nouveau pingouin à la colonie
    colonie.push_back(nouveauPingouin);

    // Étape 1 : convertir les weak_ptr en shared_ptr pour pouvoir les utiliser
    vector<shared_ptr<Pingouin>> pingouinsActifs;
    for (auto& referenceFaible : colonie) {
        if (auto pingouinActif = referenceFaible.lock()) {
            pingouinsActifs.push_back(pingouinActif);
        }
    }

    // Étape 2 : trier les pingouins par temps de parcours croissant
    sort(pingouinsActifs.begin(), pingouinsActifs.end(),
         [](const shared_ptr<Pingouin>& a, const shared_ptr<Pingouin>& b) {
             return a->calculerTempsParcours() < b->calculerTempsParcours();
         });

    // Étape 3 : reconstruire la colonie avec les références triées
    colonie.clear();
    for (auto& pingouinTrie : pingouinsActifs) {
        colonie.push_back(pingouinTrie);
    }
}
